18 asset fund_calculate_per_block(time_point_sec start_date, time_point_sec end_date, const asset& balance)
19 {
20  FC_ASSERT(start_date < end_date, "Start time ${1} must be less end time ${2}", ("1", start_date)("2", end_date));
21
