186  void update_rank()
187  {
188  auto direct = float(direct_positive_votes) / direct_votes;
189  auto indirect = float(indirect_positive_votes) / indirect_votes;
190  auto direct_order = log(direct_votes);
191  auto indirect_order = log(indirect_votes);
192 
193  if (!(direct_positive_votes + indirect_positive_votes))
194  {
195  direct_order *= -1;
196  indirect_order *= -1;
197  }
198 
199  direct *= direct;
200  indirect *= indirect;
201 
202  direct *= direct_order * 10;
203  indirect *= indirect_order;
204 
205  rank = direct + indirect;
206  }
207 };
