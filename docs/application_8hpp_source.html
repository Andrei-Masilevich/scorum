203 {
204  std::weak_ptr<C> weak_c = c.shared_from_this();
205  return sig.connect([weak_c, f](Args... args) {
206  std::shared_ptr<C> shared_c = weak_c.lock();
207  if (!shared_c)
208  return;
209  ((*shared_c).*f)(args...);
210  });
211 }
